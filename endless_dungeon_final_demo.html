import random

def roll_d20():
    """Returns a random integer between 1 and 20"""
    return random.randint(1, 20)

def roll_damage(min_dmg, max_dmg):
    """Returns a random integer between min and max"""
    return random.randint(min_dmg, max_dmg)

if __name__ == "__main__":
    print("Testing D20 Roll:", roll_d20())

import random

def get_random_monster():
    """Returns a dictionary with monster stats"""
    names = ["Goblin", "Skeleton", "Slime", "Dragon"]
    
    monster = {
        "name": random.choice(names),
        "hp": random.randint(10, 20),
        "strength": random.randint(2, 5)
    }
    return monster

if __name__ == "__main__":
    print("Testing Monster Gen:", get_random_monster())

ğŸ’° treasure.py

import random

def search_chest():
    """Returns a string item or None"""
    chance = random.random()
    
    if chance < 0.20:
        return "Potion"
    elif chance < 0.50:
        return "Gold"
    else:
        return None

if __name__ == "__main__":
    print("Testing Chest:", search_chest())
