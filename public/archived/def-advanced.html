def analyze_student_performance(grades):
    # Initialize variables - common pattern in all activities
    total_points = 0
    grade_count = 0
    passing_grades = 0
    honor_roll = False
    
    # WHILE LOOP - Process each grade (like activities 1, 2, 6, 7, 8)
    while grade_count < len(grades):
        current_grade = grades[grade_count]
        total_points += current_grade
        
        # IF/ELIF/ELSE - Decision making (like activities 2, 3, 4, 5)
        if current_grade >= 60:
            passing_grades += 1
            
            # NESTED IF - Advanced logic (activities 7, 8)
            if current_grade >= 90:
                print(f"Excellent work on assignment {grade_count + 1}!")
        else:
            print(f"Need improvement on assignment {grade_count + 1}")
        
        grade_count += 1  # CRITICAL - prevent infinite loops!
    
    # Calculate final statistics - like activities 5, 7, 8
    if grade_count > 0:
        average = total_points / grade_count
        pass_rate = (passing_grades / grade_count) * 100
        
        # Multi-condition logic - advanced decision making
        if average >= 90 and pass_rate == 100:
            honor_roll = True
            performance = "Outstanding"
        elif average >= 80:
            performance = "Good"
        elif average >= 70:
            performance = "Satisfactory"
        else:
            performance = "Needs Improvement"
    else:
        return "No grades to analyze"
    
    # RETURN formatted result - like all activities below
    return f"Average: {average:.1f}%, Pass Rate: {pass_rate:.1f}%, Performance: {performance}, Honor Roll: {honor_roll}"
