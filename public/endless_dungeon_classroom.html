# ===== MAIN.PY =====
import dice_master
import encounters
import treasure

player_hp = 50

print("‚öîÔ∏è WELCOME TO THE DUNGEON ‚öîÔ∏è")

# 1. SPAWN A MONSTER
enemy = encounters.get_random_monster()
print(f"A wild {enemy['name']} appears with {enemy['hp']} HP!")

# 2. THE COMBAT LOOP
while player_hp > 0 and enemy['hp'] > 0:
    input("Press Enter to attack...")
    
    # PLAYER ATTACKS
    roll = dice_master.roll_d20()
    
    if roll >= 10:
        dmg = dice_master.roll_damage(5, 10)
        enemy['hp'] = enemy['hp'] - dmg
        print(f"You hit for {dmg} damage!")
    else:
        print("You missed!")
        
    # MONSTER ATTACKS (If it's still alive)
    if enemy['hp'] > 0:
        player_hp -= enemy['strength']
        print(f"The monster hits you! You have {player_hp} HP left.")

# 3. END GAME
if player_hp > 0:
    print("You won!")
    loot = treasure.search_chest()
    print(f"You found: {loot}")
else:
    print("Game Over.")
