  • View Mood History (Option 2): Reads and displays all logged moods from the file

import os
import time

FILENAME = 'mood_log.txt'

def log_mood_entry():
    """Handles user input, file creation, and appending data."""
    
    # 1. Get user input
    mood = input("What is your current mood? (e.g., Happy, Stressed): ")
    timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
    
    # Format the log line, including the necessary newline character
    log_entry = f"[{timestamp}] Mood: {mood}\n"
    
    # 2. Critical Safety Check
    if not os.path.exists(FILENAME):
        # Path A: File does NOT exist -> use 'w' mode to create it and write the header
        try:
            with open(FILENAME, 'w') as f:
                f.write("--- Daily Mood Tracker Log ---\n")
                f.write(log_entry)
            print(f"\n[INFO] Created new log file and recorded your mood.")
        except Exception as e:
            print(f"\n[ERROR] Failed to create file: {e}")
    else:
        # Path B: File DOES exist -> use 'a' mode to append the new entry
        try:
            with open(FILENAME, 'a') as f:
                f.write(log_entry)
            print(f"\n[INFO] Successfully appended your mood entry.")
        except Exception as e:
            print(f"\n[ERROR] Failed to append to file: {e}")

def view_history():
    """Handles checking for file existence and reading all contents."""
    
    # 1. Safety Check (Q23) - Prevent FileNotFoundError
    if not os.path.exists(FILENAME):
        print("\n--- History Log ---")
        print(">> No history found. Log a mood entry first!")
        print("-------------------")
        return

    # 2. File exists -> Open in 'r' mode and read
    try:
        with open(FILENAME, 'r') as f:
            # f.read() loads the entire file contents as a single string
            history = f.read()
            
            print("\n--- Your Mood History ---")
            print(history.strip())
            print("-------------------------")
            
    except Exception as e:
        # Catch reading errors (e.g., file permissions changed)
        print(f"\n[ERROR] Could not read history file: {e}")


def main_menu():
    """The main application loop with robust exception handling."""
    while True:
        print("\n==============================")
        print("  Daily Mood Tracker Menu")
        print("==============================")
        print("1. Log Current Mood")
        print("2. View Mood History")
        print("3. Exit")
        
        # Use try/except for robust user input handling
        try:
            choice = input("Enter your choice (1, 2, or 3): ")
            
            if choice == '1':
                log_mood_entry()
            elif choice == '2':
                view_history()
            elif choice == '3':
                print("Thank you for using the Mood Tracker. Goodbye!")
                break
            else:
                # Handle invalid numbers or characters (Triggers ValueError conceptually)
                print("[WARNING] Invalid choice. Please enter 1, 2, or 3.")

        except Exception as e:
            # Catch unexpected exceptions in the loop
            print(f"[FATAL ERROR] An unexpected error occurred: {e}. Restarting menu...")
            
# Run the application
main_menu()

üîç Key Concepts to Study:

  • os.path.exists(FILENAME): Returns True if file exists, False if not - prevents crashes
  • File Modes: 'w' creates/overwrites, 'a' appends to end, 'r' reads existing file
  • with open() as f: Context manager automatically closes file after use
  • try/except blocks: Catch errors gracefully instead of crashing the program
  • f.write() vs f.read(): Writing adds text to file, reading retrieves all content
  • String formatting: f-strings like f"[{timestamp}]" embed variables in text
  • time.strftime(): Formats current time as readable string
  • Conditional branching: if/else paths handle different file states
