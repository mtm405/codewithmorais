üìò New Concepts: Working with Files

The os Module

The os module provides functions for interacting with the operating system, including file and directory operations.

üìå Key Function: os.path.exists(filename)
  • Returns True if the file exists
  • Returns False if the file does not exist
  • Helps prevent errors when trying to open files that might not be there
import os

# Check if a file exists before opening it
if os.path.exists("quiz_data.json"):
    print("File found!")
else:
    print("File not found. Using defaults.")

The json Module

JSON (JavaScript Object Notation) is a popular format for storing and exchanging data. Python's json module lets you convert Python data structures (like lists and dictionaries) to JSON format and back.

üìå Key Functions:
  • json.load(file) - Reads JSON data from a file and converts it to Python data structures
  • json.dump(data, file) - Converts Python data structures to JSON and writes to a file
  • json.loads(string) - Converts a JSON string to Python data

Reading from a JSON File:

import json

# Open and read from a JSON file
with open("quiz_data.json", "r") as file:
    data = json.load(file)
    print(data)  # Now you have a Python list or dictionary!

Writing to a JSON File:

import json

questions = [
    {"question": "What is 2+2?", "answer": "4"},
    {"question": "What is Python?", "answer": "A programming language"}
]

# Save to a JSON file
with open("quiz_data.json", "w") as file:
    json.dump(questions, file, indent=4)  # indent makes it pretty!
üí° Why JSON? JSON files are human-readable, easy to edit, and work across different programming languages. They're perfect for storing quiz questions, game data, user settings, and more!

Complete Starter Template

import json
import os  # Used to check if the quiz data file exists

# --- GLOBAL CONFIGURATION ---
QUIZ_FILE = "quiz_data.json"

# --- STARTER DATA (Used if the file does not exist) ---
STARTER_QUESTIONS = [
    {
        "question": "What is the result of 5 % 2 in Python?",
        "options": ["A) 2.5", "B) 1", "C) 2", "D) 0"],
        "correct_answer": "B"
    },
    {
        "question": "Which loop is best used when you know exactly how many times you want to iterate?",
        "options": ["A) while", "B) for", "C) do-while", "D) infinite"],
        "correct_answer": "B"
    },
    {
        "question": "How do you define a function in Python?",
        "options": ["A) function my_func():", "B) define my_func():", "C) def my_func():", "D) start my_func():"],
        "correct_answer": "C"
    }
]

# --- PHASE 1: Data Handling Functions ---

def load_quiz():
    """
    Loads questions from the JSON file. If the file is not found,
    it returns the STARTER_QUESTIONS list.
    """
    try:
        # STUDENT TASK: Use the 'with open' structure and json.load() here.
        # If the file exists, load and return the data.
        if os.path.exists(QUIZ_FILE):
            with open(QUIZ_FILE, 'r') as file:
                file_content = file.read().strip()
                if file_content:
                    questions = json.loads(file_content)
                    print(f"Loaded {len(questions)} questions from {QUIZ_FILE}.")
                    return questions
                else:
                    print("Quiz file is empty. Using starter questions.")
                    return STARTER_QUESTIONS
        else:
            print("Quiz file not found. Using starter questions.")
            return STARTER_QUESTIONS
    except json.JSONDecodeError:
        print(f"Error reading {QUIZ_FILE}. Using starter questions.")
        return STARTER_QUESTIONS
    except Exception as e:
        print(f"Unexpected error: {e}. Using starter questions.")
        return STARTER_QUESTIONS

def save_quiz(questions):
    """
    Saves the complete list of questions back to the JSON file.
    """
    # STUDENT TASK: Use 'with open' and json.dump() here.
    # Ensure you open the file in 'w' (write) mode.
    try:
        with open(QUIZ_FILE, 'w') as file:
            json.dump(questions, file, indent=4)
        print(f"\nSuccessfully saved {len(questions)} questions to {QUIZ_FILE}.")
    except Exception as e:
        print(f"\nERROR: Could not save quiz data. {e}")


# --- PHASE 2: Quiz Engine Functions ---

def run_quiz(questions):
    """
    Runs the quiz, iterates through questions, collects answers, 
    and tracks the score.
    """
    # STUDENT TASK: Initialize score and write the main quiz loop logic here.
    
    score = 0
    total_questions = len(questions)

    if total_questions == 0:
        print("\nNo questions available! Try creating some first!")
        return

    print("\n--- Starting Quiz ---")
    
    for i, q in enumerate(questions):
        print(f"\nQuestion {i + 1}/{total_questions}: {q['question']}")
        
        # Print options
        for option in q['options']:
            print(f"  {option}")

        # Get user input
        while True:
            user_answer = input("Your answer (e.g., A, B, C, D): ").strip().upper()
            
            # Validation check
            valid_options = [opt[0] for opt in q['options'] if opt[0].isalpha()]
            if user_answer in valid_options:
                break
            else:
                print("Invalid input. Please choose a valid option.")

        # Check answer using conditional logic
        if user_answer == q['correct_answer'].upper():
            print("‚úì Correct!")
            score += 1
        else:
            print(f"‚úó Incorrect. The correct answer was: {q['correct_answer']}")

    # After the loop, display the final score
    display_score(score, total_questions)

def display_score(score, total):
    """
    Calculates and displays the final score and percentage.
    """
    # STUDENT TASK: Add conditional logic (if/elif/else) for feedback.
    
    percent = (score / total) * 100 if total > 0 else 0
    print("\n" + "="*40)
    print("--- Quiz Results ---")
    print(f"You answered {score} out of {total} questions correctly.")
    print(f"Final Score: {percent:.2f}%")

    if percent >= 90:
        print("üéâ Outstanding! You are a Python master!")
    elif percent >= 70:
        print("üëç Good job! You have a solid understanding.")
    elif percent >= 50:
        print("üìñ Keep studying to improve your score.")
    else:
        print("üí° Time to review the basics!")
    print("="*40)


# --- PHASE 3: Quiz Editor Function ---

def create_question():
    """
    Prompts the user for details and creates a new question dictionary.
    """
    # STUDENT TASK: Use multiple input() calls to gather all data.
    
    print("\n--- New Question Creator ---")
    
    q_text = input("Enter the question text: ").strip()
    
    # Loop to gather options
    options = []
    option_letters = ['A', 'B', 'C', 'D']
    print("Enter the 4 multiple-choice options:")
    for letter in option_letters:
        option_text = input(f"Option {letter}: ").strip()
        options.append(f"{letter}) {option_text}")
        
    # Get the correct answer
    while True:
        correct_ans = input("Enter the letter of the correct answer (A, B, C, or D): ").strip().upper()
        if correct_ans in option_letters:
            break
        else:
            print("Invalid choice. Please use A, B, C, or D.")

    new_question = {
        "question": q_text,
        "options": options,
        "correct_answer": correct_ans
    }
    
    print("\n‚úì Question created successfully!")
    return new_question

# --- PHASE 4: Main Application Flow ---

def main_menu():
    """
    The main control function that displays the menu and handles user choices.
    Uses a while loop to keep the program running.
    """
    questions = load_quiz()

    while True:
        print("\n" + "="*40)
        print("  WELCOME TO PYQUIZ")
        print("="*40)
        print("1. Take Quiz")
        print("2. Create New Question")
        print("3. Exit")
        print("="*40)
        
        # STUDENT TASK: Use input() and if/elif/else to handle choices.
        
        choice = input("Enter your choice (1, 2, or 3): ").strip()

        if choice == '1':
            run_quiz(questions)
        elif choice == '2':
            new_q = create_question()
            questions.append(new_q)
            save_quiz(questions)  # Save after adding
        elif choice == '3':
            print("\nThank you for using PyQuiz! Goodbye.")
            break
        else:
            print("Invalid choice. Please enter 1, 2, or 3.")

# Check for the typical Python entry point
if __name__ == "__main__":
    main_menu()

üí° Tips and Best Practices

üéØ Development Strategy:
  • Start Small: Get one function working before moving to the next
  • Test Often: Run your code after every few lines to catch errors early
  • Use Print Statements: Add print() statements to debug and see what's happening
  • Read Error Messages: Python's error messages tell you exactly what's wrong and where
üêõ Common Pitfalls to Avoid:

  • Not using with open() - this automatically closes files

  • Not validating user input (what if they type 'Z' when only A-D are valid?)
